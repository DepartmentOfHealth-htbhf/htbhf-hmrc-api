package uk.gov.dhsc.htbhf.hmrc;

@SpringBootApplication
@EnableSwagger2
@Import(CommonRestConfiguration.class)

    public static void main(String[] args) {
        SpringApplication.run(HMRCApplication.class, args);
    }

}
